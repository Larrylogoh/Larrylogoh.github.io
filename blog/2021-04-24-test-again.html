def isValidWalk(walk):
    n = walk.count('n')
    e = walk.count('e')
    s = walk.count('s')
    w = walk.count('w')
    position = [n-s, w-e]
    while len(walk) == 10:
        if position == [0,0]:
            return True
        else:
            return False
    else:
        return False
